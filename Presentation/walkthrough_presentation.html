agged_data <- read_and_agg(datpath = ewr_results, 
           type = 'summary',
           geopath = bom_basin_gauges,
           causalpath = causal_ewr,
           groupers = 'scenario',
           aggCols = 'ewr_achieved',
           aggsequence = aggseq,
           funsequence = funseq,
           saveintermediate = TRUE,
           namehistory = FALSE,
           keepAllPolys = FALSE,
           returnList = TRUE,
           savepath = NULL)
